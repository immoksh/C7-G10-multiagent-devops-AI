{raw_excerpt}

{{
  "severity": "<critical|warning|info>",
  "root_cause": "<short snake_case identifier, e.g. pod_crashloop, upstream_timeout, db_connection_refused>",
  "confidence": <float between 0.0 and 1.0>,
